// ========== 画面切り替え ==========
function showScreen(id){
  document.getElementById('screen-home').style.display='none';
  document.getElementById('screen-main').style.display='none';
  document.getElementById('screen-guide').style.display='none';
  document.getElementById(id).style.display='block';
}
function openMain(){showScreen('screen-main');}
function openHome(){showScreen('screen-home');}
function openGuide(){showScreen('screen-guide');}

// ========== テーマ・年齢設定 ==========
function setTheme(type){
  const themes=[
    'theme-dark','theme-light','theme-a90','theme-red','theme-blue','theme-green',
    'theme-yellow','theme-purple','theme-pink','theme-orange','theme-cyan',
    'theme-gray','theme-brown','theme-silver','theme-gold','theme-rainbow','theme-sky'
  ];
  themes.forEach(t=>document.body.classList.remove(t));
  document.body.classList.add('theme-'+type);
}
function setAge(mode){
  document.body.classList.remove('age-child','age-teen','age-adult');
  if(mode==='child') document.body.classList.add('age-child');
  else if(mode==='adult') document.body.classList.add('age-adult');
  else document.body.classList.add('age-teen');
}

// ========== セクション開閉 ==========
function toggleSection(id){
  const el=document.getElementById(id);
  if(!el) return;
  el.style.display = (el.style.display==='block') ? 'none' : 'block';
}

// ========== 共通リスト生成 ==========
function createItemRow(item){
  const row = document.createElement('div');
  row.className='item-row';

  const main = document.createElement('div');
  main.className='item-main';

  const dot = document.createElement('div');
  dot.className='dot '+(item.dot||'dot-light');
  main.appendChild(dot);

  const name = document.createElement('div');
  name.className='item-name';
  name.textContent=item.name;
  main.appendChild(name);

  if(item.tag){
    const tag=document.createElement('div');
    tag.className='item-tag';
    tag.textContent='['+item.tag+']';
    main.appendChild(tag);
  }

  row.appendChild(main);

  const btns=document.createElement('div');
  btns.className='item-buttons';

  const detailBtn=document.createElement('button');
  detailBtn.className='btn btn-main';
  detailBtn.textContent='詳細';
  detailBtn.onclick=function(){
    openDetail(item);
  };
  btns.appendChild(detailBtn);

  const punishBtn=document.createElement('button');
  punishBtn.className='btn btn-danger';
  punishBtn.textContent='罰';
  punishBtn.onclick=function(){
    row.classList.add('fade-out');
    setTimeout(()=>row.remove(),250);
  };
  btns.appendChild(punishBtn);

  row.appendChild(btns);
  return row;
}

function fillSection(list, containerId){
  const cont=document.getElementById(containerId);
  if(!cont) return;
  cont.innerHTML='';
  list.forEach(item=>{
    cont.appendChild(createItemRow(item));
  });
}

// ========== 詳細モーダル ==========
function openDetail(item){
  const modal=document.getElementById('detail-modal');
  const win=document.getElementById('detail-window');
  const tag=document.getElementById('detail-tag');
  const title=document.getElementById('detail-title');
  const body=document.getElementById('detail-body');

  win.classList.remove('detail-right','detail-mode','detail-tech','detail-pet','detail-item','detail-event','detail-badge','detail-spell');
  if(item.cat==='right') win.classList.add('detail-right');
  else if(item.cat==='mode') win.classList.add('detail-mode');
  else if(item.cat==='tech') win.classList.add('detail-tech');
  else if(item.cat==='pet') win.classList.add('detail-pet');
  else if(item.cat==='item') win.classList.add('detail-item');
  else if(item.cat==='event') win.classList.add('detail-event');
  else if(item.cat==='badge') win.classList.add('detail-badge');
  else if(item.cat==='spell') win.classList.add('detail-spell');

  tag.textContent=item.tag?('['+item.tag+']'):'';
  title.textContent=item.name;
  body.textContent=item.desc || (item.name+" に関する説明テキスト。");

  modal.style.display='flex';
}
function closeDetail(){
  document.getElementById('detail-modal').style.display='none';
}

// ========== A-90 パワー 1〜1000% ==========
function describePower(v){
  const n=Number(v);
  if(n<=50)   return "ほぼ休憩中のA-90。ノイズも小さく、ただの気配レベル。";
  if(n<=150)  return "標準のA-90。ときどきノイズで驚かせてくる通常モード。";
  if(n<=300)  return "じわっと本気を出し始めたA-90。接近速度やノイズ頻度が上がる。";
  if(n<=500)  return "半暴走A-90。視界バグや挙動がかなり危険な領域に入る。";
  if(n<=700)  return "高負荷モードのA-90。回避には本気の集中が必要。";
  if(n<=900)  return "ほぼフルパワーのA-90。少しの油断で即終了級。";
  if(n<1000)  return "限界突破寸前A-90。世界のほうが耐えられているか怪しい。";
  return "1000% OVERDRIVE A-90。仕様を完全にぶち破った暴走状態。";
}
function updatePower(){
  const slider=document.getElementById('power-range');
  const label=document.getElementById('power-label');
  const fill=document.getElementById('power-fill');
  const state=document.getElementById('power-state');
  const v=Number(slider.value);
  label.textContent=v+"%";
  let gauge=v/10; if(gauge>100) gauge=100;
  fill.style.width=gauge+"%";
  state.textContent=describePower(v);
}
